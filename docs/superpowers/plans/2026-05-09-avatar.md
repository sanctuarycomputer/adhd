# Avatar + AvatarGroup Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a "blue-chip" Avatar + AvatarGroup component pair as a reference example for the codebase, demonstrating the mix of Tailwind v4 utilities with selectively hard-coded pixel values.

**Architecture:** Two sibling component folders, each with `index.tsx`. Avatar is fully standalone. AvatarGroup imports Avatar (and a few size-token lookups Avatar exports) and uses `React.cloneElement` to push `size` down to its Avatar children, applying a separating ring via a Tailwind descendant selector. Variant tokens live in module-scope lookup objects to avoid per-render allocation.

**Tech Stack:** Next.js 16, React 19, Tailwind CSS v4, TypeScript (strict), `next/image`.

**Reference:** Spec at `docs/superpowers/specs/2026-05-09-avatar-design.md`.

```tsx
export type AvatarSize = "xs" | "sm" | "md" | "lg" | "xl";
export type AvatarShape = "circle" | "square";
export type AvatarStatus = "online" | "away" | "offline";

export interface AvatarProps {
  name: string;
  src?: string;
  size?: AvatarSize;
  shape?: AvatarShape;
  status?: AvatarStatus;
  className?: string;
}

export const SIZE_BOX: Record<AvatarSize, string> = {
  xs: "h-6 w-6",
  sm: "h-8 w-8",
  md: "h-10 w-10",
  lg: "h-12 w-12",
  xl: "h-16 w-16",
};

export const SIZE_TEXT: Record<AvatarSize, string> = {
  xs: "text-[10px]",
  sm: "text-xs",
  md: "text-sm",
  lg: "text-base",
  xl: "text-lg",
};

export const SHAPE: Record<AvatarShape, Record<AvatarSize, string>> = {
  circle: {
    xs: "rounded-full",
    sm: "rounded-full",
    md: "rounded-full",
    lg: "rounded-full",
    xl: "rounded-full",
  },
  square: {
    xs: "rounded-md",
    sm: "rounded-md",
    md: "rounded-lg",
    lg: "rounded-lg",
    xl: "rounded-lg",
  },
};

const PALETTE = [
  "bg-zinc-200 text-zinc-700 dark:bg-zinc-800 dark:text-zinc-200",
  "bg-stone-200 text-stone-700 dark:bg-stone-800 dark:text-stone-200",
  "bg-neutral-300 text-neutral-800 dark:bg-neutral-700 dark:text-neutral-100",
  "bg-slate-200 text-slate-700 dark:bg-slate-800 dark:text-slate-200",
  "bg-gray-200 text-gray-700 dark:bg-gray-800 dark:text-gray-200",
  "bg-zinc-300 text-zinc-800 dark:bg-zinc-700 dark:text-zinc-100",
];

function getInitials(name: string): string {
  const parts = name.trim().split(/\s+/).slice(0, 2);
  return parts.map((p) => p[0]?.toUpperCase() ?? "").join("");
}

function getPaletteIndex(name: string): number {
  let sum = 0;
  for (let i = 0; i < name.length; i++) sum += name.charCodeAt(i);
  return sum % PALETTE.length;
}

export function Avatar({
  name,
  size = "md",
  shape = "circle",
  className = "",
}: AvatarProps) {
  const box = SIZE_BOX[size];
  const radius = SHAPE[shape][size];
  const palette = PALETTE[getPaletteIndex(name)];

  return (
    <span className={`relative inline-block ${box} ${radius} ${className}`}>
      <span
        className={`flex h-full w-full items-center justify-center overflow-hidden font-medium uppercase select-none ${radius} ${SIZE_TEXT[size]} ${palette}`}
        aria-label={name}
      >
        {getInitials(name)}
      </span>
    </span>
  );
}
```

Create `app/components/avatar-group/index.tsx` with this content:

```tsx
import {
  Children,
  cloneElement,
  isValidElement,
  type ReactElement,
  type ReactNode,
} from "react";
import { Avatar, type AvatarSize } from "../avatar";

export interface AvatarGroupProps {
  children: ReactNode;
  size?: AvatarSize;
  className?: string;
}

const OVERLAP: Record<AvatarSize, string> = {
  xs: "-space-x-1",
  sm: "-space-x-1",
  md: "-space-x-2",
  lg: "-space-x-2",
  xl: "-space-x-3",
};

export function AvatarGroup({
  children,
  size = "md",
  className = "",
}: AvatarGroupProps) {
  const items = Children.toArray(children).map((child) => {
    if (isValidElement(child) && child.type === Avatar) {
      return cloneElement(child as ReactElement<{ size?: AvatarSize }>, {
        size,
      });
    }
    return child;
  });

  return (
    <div
      className={`flex items-center ${OVERLAP[size]} [&>*]:ring-2 [&>*]:ring-white dark:[&>*]:ring-zinc-950 ${className}`}
    >
      {items}
    </div>
  );
}
```

Replace the `AvatarGroup` function with:

```tsx
export function AvatarGroup({
  children,
  size = "md",
  max,
  className = "",
}: AvatarGroupProps) {
  const all = Children.toArray(children);
  const truncated = max != null && all.length > max;
  const visible = truncated ? all.slice(0, max) : all;
  const overflow = truncated ? all.length - max : 0;

  const items = visible.map((child) => {
    if (isValidElement(child) && child.type === Avatar) {
      return cloneElement(child as ReactElement<{ size?: AvatarSize }>, {
        size,
      });
    }
    return child;
  });

  return (
    <div
      className={`flex items-center ${OVERLAP[size]} [&>*]:ring-2 [&>*]:ring-white dark:[&>*]:ring-zinc-950 ${className}`}
    >
      {items}
      {overflow > 0 && (
        <span
          className={`relative inline-flex items-center justify-center font-medium ${SIZE_BOX[size]} ${SHAPE.circle[size]} ${SIZE_TEXT[size]} bg-zinc-200 text-zinc-700 dark:bg-zinc-800 dark:text-zinc-200`}
          aria-label={`${overflow} more`}
        >
          +{overflow}
        </span>
      )}
    </div>
  );
}
```

Note: the overflow tile is always `circle`-shaped regardless of children's shapes — a v1 simplification. If a future need arises for square overflow tiles, add a `shape` prop to AvatarGroup.
